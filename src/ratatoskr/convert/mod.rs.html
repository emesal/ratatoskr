1//! Conversions between ratatoskr types and llm crate types.
2//!
3//! This module is internal and handles the translation layer between
4//! our stable public types and the llm crate's internal types.
5
6use llm::chat::ChatMessage as LlmMessage;
7
8use crate::{Message, MessageContent, Role, ToolCall, Usage as RataUsage};
9
10/// Extract text content from a message, returning an error for non-text variants.
11///
12/// `MessageContent` is `#[non_exhaustive]` and will gain variants (e.g. `Image`,
13/// `MultiPart`). The wildcard arm ensures graceful failure when that happens.
14fn extract_text(content: &MessageContent, role: &str) -> crate::Result<String> {
15    match content {
16        MessageContent::Text(text) => Ok(text.clone()),
17        #[allow(unreachable_patterns)]
18        other => Err(crate::RatatoskrError::InvalidInput(format!(
19            "{role} message content type not supported for llm conversion: {other:?}"
20        ))),
21    }
22}
23
24/// Convert our messages to llm crate messages
25pub fn to_llm_messages(messages: &[Message]) -> crate::Result<(Option<String>, Vec<LlmMessage>)> {
26    let mut system_prompt = None;
27    let mut llm_messages = Vec::with_capacity(messages.len());
28
29    for msg in messages {
30        match &msg.role {
31            Role::System => {
32                let text = extract_text(&msg.content, "system")?;
33                system_prompt = Some(text);
34            }
35            Role::User => {
36                let text = extract_text(&msg.content, "user")?;
37                llm_messages.push(LlmMessage::user().content(text).build());
38            }
39            Role::Assistant => {
40                if let Some(tool_calls) = &msg.tool_calls {
41                    // Assistant message with tool calls
42                    let llm_tool_calls: Vec<llm::ToolCall> = tool_calls
43                        .iter()
44                        .map(|tc| llm::ToolCall {
45                            id: tc.id.clone(),
46                            call_type: "function".to_string(),
47                            function: llm::FunctionCall {
48                                name: tc.name.clone(),
49                                arguments: tc.arguments.clone(),
50                            },
51                        })
52                        .collect();
53
54                    let content = msg.content.as_text().unwrap_or_default();
55                    llm_messages.push(
56                        LlmMessage::assistant()
57                            .tool_use(llm_tool_calls)
58                            .content(content)
59                            .build(),
60                    );
61                } else {
62                    let text = extract_text(&msg.content, "assistant")?;
63                    llm_messages.push(LlmMessage::assistant().content(text).build());
64                }
65            }
66            Role::Tool { tool_call_id } => {
67                let text = extract_text(&msg.content, "tool")?;
68                llm_messages.push(
69                    LlmMessage::user()
70                        .tool_result(vec![llm::ToolCall {
71                            id: tool_call_id.clone(),
72                            call_type: "function".to_string(),
73                            function: llm::FunctionCall {
74                                name: String::new(), // Not needed for result
75                                arguments: text,
76                            },
77                        }])
78                        .build(),
79                );
80            }
81        }
82    }
83
84    Ok((system_prompt, llm_messages))
85}

87/// Convert llm crate tool calls to our format
88pub fn from_llm_tool_calls(calls: &[llm::ToolCall]) -> Vec<ToolCall> {
89    calls
90        .iter()
91        .map(|c| ToolCall {
92            id: c.id.clone(),
93            name: c.function.name.clone(),
94            arguments: c.function.arguments.clone(),
95        })
96        .collect()
97}

99/// Convert llm crate usage to our format
100pub fn from_llm_usage(usage: &llm::chat::Usage) -> RataUsage {
101    RataUsage {
102        prompt_tokens: usage.prompt_tokens as u64,
103        completion_tokens: usage.completion_tokens as u64,
104        total_tokens: (usage.prompt_tokens + usage.completion_tokens) as u64,
105        reasoning_tokens: None, // llm crate doesn't expose this yet
106    }
107}

109#[cfg(test)]
110mod tests {
111    use super::*;
112
113    #[test]
114    fn system_message_extracted_as_system_prompt() {
115        let messages = vec![Message::system("you are helpful"), Message::user("hello")];
116        let (system, llm_msgs) = to_llm_messages(&messages).unwrap();
117        assert_eq!(system, Some("you are helpful".to_string()));
118        assert_eq!(llm_msgs.len(), 1);
119    }
120
121    #[test]
122    fn no_system_message_returns_none() {
123        let messages = vec![Message::user("hello")];
124        let (system, llm_msgs) = to_llm_messages(&messages).unwrap();
125        assert!(system.is_none());
126        assert_eq!(llm_msgs.len(), 1);
127    }
128
129    #[test]
130    fn multi_role_conversation() {
131        let messages = vec![
132            Message::system("be helpful"),
133            Message::user("hi"),
134            Message::assistant("hello!"),
135            Message::user("how are you?"),
136        ];
137        let (system, llm_msgs) = to_llm_messages(&messages).unwrap();
138        assert_eq!(system, Some("be helpful".to_string()));
139        assert_eq!(llm_msgs.len(), 3); // user, assistant, user (system extracted)
140    }
141
142    #[test]
143    fn tool_call_conversion() {
144        let tool_calls = vec![ToolCall {
145            id: "call_123".to_string(),
146            name: "get_weather".to_string(),
147            arguments: r#"{"city":"london"}"#.to_string(),
148        }];
149        let messages = vec![Message::assistant_with_tool_calls(None, tool_calls)];
150        let (_, llm_msgs) = to_llm_messages(&messages).unwrap();
151        assert_eq!(llm_msgs.len(), 1);
152    }
153
154    #[test]
155    fn tool_result_message() {
156        let messages = vec![Message::tool_result("call_123", r#"{"temp": 15}"#)];
157        let (_, llm_msgs) = to_llm_messages(&messages).unwrap();
158        assert_eq!(llm_msgs.len(), 1);
159    }
160
161    #[test]
162    fn from_llm_tool_calls_roundtrip() {
163        let llm_calls = vec![llm::ToolCall {
164            id: "call_1".to_string(),
165            call_type: "function".to_string(),
166            function: llm::FunctionCall {
167                name: "search".to_string(),
168                arguments: r#"{"q":"rust"}"#.to_string(),
169            },
170        }];
171
172        let ours = from_llm_tool_calls(&llm_calls);
173        assert_eq!(ours.len(), 1);
174        assert_eq!(ours[0].id, "call_1");
175        assert_eq!(ours[0].name, "search");
176        assert_eq!(ours[0].arguments, r#"{"q":"rust"}"#);
177    }

179    #[test]

181        let llm_usage = llm::chat::Usage {

184            total_tokens: 150,
185            completion_tokens_details: None,
186            prompt_tokens_details: None,

188        let usage = from_llm_usage(&llm_usage);
189        assert_eq!(usage.prompt_tokens, 100);
190        assert_eq!(usage.completion_tokens, 50);

192        assert!(usage.reasoning_tokens.is_none());
193    }
