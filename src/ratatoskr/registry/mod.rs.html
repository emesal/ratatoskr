1//! Model registry — centralised model metadata with layered merge.
2//!
3//! The registry holds [`ModelMetadata`] entries from multiple sources:
4//! 1. **Embedded seed** — compiled-in JSON, always available
5//! 2. **Cached remote** — curated data from `emesal/ratatoskr-registry`
6//! 3. **Live provider data** — runtime API queries
7//!
8//! Merge priority: later data overrides earlier (live > remote > embedded).
9//! Within a merge, per-parameter overrides are applied (not whole-entry replacement).
10//!
11//! See [`remote`] for the fetch/cache mechanism.
12
13pub mod preset;
14pub mod remote;
15
16pub use preset::{PresetEntry, PresetParameters, PresetResolution};

18use std::collections::{BTreeMap, BTreeSet, HashMap};

20use tracing::{info, warn};

22use crate::{ModelCapability, ModelMetadata};

24/// Centralised model metadata registry.
25///
26/// Thread-safe reads are ensured by storing data in a `HashMap` keyed by model ID.
27/// For concurrent access (e.g., from `EmbeddedGateway`), wrap in `Arc<RwLock<>>`.
28#[derive(Debug, Clone, Default)]
29pub struct ModelRegistry {
30    entries: HashMap<String, ModelMetadata>,
31    /// Autoconfig presets: `(cost_tier, capability_string) → preset entry`.
32    presets: BTreeMap<String, BTreeMap<String, PresetEntry>>,
33}
34
35impl ModelRegistry {
36    /// Create an empty registry.
37    pub fn new() -> Self {
38        Self::default()
39    }
40
41    /// Insert a model entry, replacing any existing entry with the same ID.
42    pub fn insert(&mut self, metadata: ModelMetadata) {
43        self.entries.insert(metadata.info.id.clone(), metadata);
44    }
45
46    /// Get metadata for a model by ID.
47    pub fn get(&self, model: &str) -> Option<&ModelMetadata> {
48        self.entries.get(model)
49    }
50
51    /// List all model metadata entries.
52    pub fn list(&self) -> Vec<&ModelMetadata> {
53        self.entries.values().collect()
54    }
55
56    /// Merge a single entry: update existing parameters or insert new.
57    ///
58    /// If the model already exists, parameters from `incoming` override the
59    /// existing entry's parameters (per-key), and scalar fields (pricing,
60    /// max_output_tokens, context_window) are replaced if present in `incoming`.
61    /// If the model doesn't exist, it's inserted directly.
62    pub fn merge(&mut self, incoming: ModelMetadata) {
63        let id = incoming.info.id.clone();
64        if let Some(existing) = self.entries.get_mut(&id) {
65            // Merge parameters (incoming overrides existing per-key).
66            for (name, avail) in incoming.parameters {
67                existing.parameters.insert(name, avail);
68            }
69            // Scalar overrides: only replace if incoming has a value.
70            if incoming.pricing.is_some() {
71                existing.pricing = incoming.pricing;
72            }
73            if incoming.max_output_tokens.is_some() {
74                existing.max_output_tokens = incoming.max_output_tokens;
75            }
76            if incoming.info.context_window.is_some() {
77                existing.info.context_window = incoming.info.context_window;
78            }
79            // Merge capabilities (union).
80            for cap in incoming.info.capabilities {
81                if !existing.info.capabilities.contains(&cap) {
82                    existing.info.capabilities.push(cap);
83                }
84            }
85        } else {
86            self.entries.insert(id, incoming);
87        }
88    }
89
90    /// Merge a batch of entries.
91    pub fn merge_batch(&mut self, batch: Vec<ModelMetadata>) {
92        for entry in batch {
93            self.merge(entry);
94        }
95    }
96
97    /// Filter entries by capability.
98    pub fn filter_by_capability(&self, capability: ModelCapability) -> Vec<&ModelMetadata> {
99        self.entries
100            .values()
101            .filter(|m| m.info.capabilities.contains(&capability))
102            .collect()
103    }
104
105    /// Number of entries in the registry.
106    pub fn len(&self) -> usize {
107        self.entries.len()
108    }
109
110    /// Whether the registry is empty.
111    pub fn is_empty(&self) -> bool {
112        self.entries.is_empty()
113    }
114
115    // ===== Presets =====
116
117    /// Look up a preset entry for the given tier and capability.
118    pub fn preset(&self, tier: &str, capability: &str) -> Option<&PresetEntry> {
119        self.presets
120            .get(tier)
121            .and_then(|cap_map| cap_map.get(capability))
122    }
123
124    /// Get all presets for a cost tier.
125    pub fn presets_for_tier(&self, tier: &str) -> Option<&BTreeMap<String, PresetEntry>> {
126        self.presets.get(tier)
127    }
128
129    /// Return all preset tiers and their capability names (no model details).
130    pub fn all_preset_keys(&self) -> BTreeMap<String, BTreeSet<String>> {
131        self.presets
132            .iter()
133            .map(|(tier, caps)| (tier.clone(), caps.keys().cloned().collect()))
134            .collect()
135    }
136
137    /// Insert or update a single preset.
138    pub fn set_preset(&mut self, tier: &str, capability: &str, entry: PresetEntry) {
139        self.presets
140            .entry(tier.to_owned())
141            .or_default()
142            .insert(capability.to_owned(), entry);
143    }
144
145    /// Merge incoming presets (incoming overrides existing per-key).
146    pub fn merge_presets(&mut self, incoming: BTreeMap<String, BTreeMap<String, PresetEntry>>) {
147        for (tier, cap_map) in incoming {
148            let existing = self.presets.entry(tier).or_default();
149            for (capability, entry) in cap_map {
150                existing.insert(capability, entry);
151            }
152        }
153    }
154
155    /// Warn about preset model IDs that aren't present in the registry entries.
156    ///
157    /// This is advisory — the model may still be resolvable at runtime via provider APIs.
158    fn validate_presets(&self) {
159        for (tier, cap_map) in &self.presets {
160            for (capability, entry) in cap_map {
161                let model_id = entry.model();
162                if !self.entries.contains_key(model_id) {
163                    warn!(
164                        tier = %tier,
165                        capability = %capability,
166                        model_id = %model_id,
167                        "preset references model not in registry"
168                    );
169                }
170            }
171        }
172    }
173
174    /// Load cached remote registry data and merge into this registry.
175    ///
176    /// Reads from the local cache file only (no network I/O). If the file
177    /// is missing or corrupt, this is a no-op (logged at warn level).
178    pub fn with_cached_remote(mut self, path: &std::path::Path) -> Self {
179        if let Some(payload) = remote::load_cached(path) {
180            info!(
181                count = payload.models.len(),
182                "loaded cached remote registry"
183            );
184            self.merge_batch(payload.models);
185            self.merge_presets(payload.presets);
186            self.validate_presets();
187        }
188        self
189    }
190
191    /// Create a registry pre-populated with the embedded seed data.
192    ///
193    /// The seed contains a curated set of well-known models with parameter
194    /// metadata, pricing, and capabilities. It's always available as a
195    /// fallback when live provider APIs are unreachable.
196    pub fn with_embedded_seed() -> Self {
197        let mut registry = Self::new();
198        match crate::registry::remote::parse_payload(EMBEDDED_SEED) {
199            Ok(payload) => {
200                for entry in payload.models {
201                    registry.insert(entry);
202                }
203                registry.merge_presets(payload.presets);
204                registry.validate_presets();
205            }
206            Err(e) => {
207                // This should never happen — seed is compiled in and tested.
208                // Log the error but don't panic; an empty registry is usable.
209                warn!(error = %e, "failed to parse embedded model seed");
210            }
211        }
212        registry
213    }
214}
215
216/// Raw JSON seed data compiled into the binary.
217const EMBEDDED_SEED: &str = include_str!("seed.json");
