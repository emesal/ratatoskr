9/// What capabilities a gateway supports.
10#[derive(Debug, Clone, Default, PartialEq, Eq, Serialize, Deserialize)]
11pub struct Capabilities {
12    /// Multi-turn chat conversations.
13    pub chat: bool,
14    /// Streaming chat responses.
15    pub chat_streaming: bool,
16    /// Single-turn text generation.
17    pub generate: bool,
18    /// Tool/function calling support.
19    pub tool_use: bool,
20    /// Text embeddings.
21    #[serde(alias = "embeddings")]
22    pub embed: bool,
23    /// Natural language inference.
24    pub nli: bool,
25    /// Zero-shot classification.
26    #[serde(alias = "classification")]
27    pub classify: bool,
28    /// Stance detection toward target topics.
29    pub stance: bool,
30    /// Token counting for models.
31    pub token_counting: bool,
32    /// Local inference (no API calls needed).
33    pub local_inference: bool,
34}
35
36impl Capabilities {
37    /// Phase 1 embedded gateway capabilities (chat only)
38    pub fn chat_only() -> Self {
39        Self {
40            chat: true,
41            chat_streaming: true,
42            generate: true,
43            tool_use: true,
44            ..Default::default()
45        }
46    }
47
48    /// Full capabilities (all features).
49    pub fn full() -> Self {
50        Self {
51            chat: true,
52            chat_streaming: true,
53            generate: true,
54            tool_use: true,
55            embed: true,
56            nli: true,
57            classify: true,
58            stance: true,
59            token_counting: true,
60            local_inference: true,
61        }
62    }
63
64    /// Local inference only capabilities (no API needed)
65    pub fn local_only() -> Self {
66        Self {
67            embed: true,
68            nli: true,
69            token_counting: true,
70            local_inference: true,
71            ..Default::default()
72        }
73    }
74
75    /// HuggingFace API capabilities (embeddings, NLI, classification)
76    pub fn huggingface_only() -> Self {
77        Self {
78            embed: true,
79            nli: true,
80            classify: true,
81            ..Default::default()
82        }
83    }
84
85    /// Merge capabilities using OR logic (combines two capability sets).
86    pub fn merge(&self, other: &Self) -> Self {
87        Self {
88            chat: self.chat || other.chat,
89            chat_streaming: self.chat_streaming || other.chat_streaming,
90            generate: self.generate || other.generate,
91            tool_use: self.tool_use || other.tool_use,
92            embed: self.embed || other.embed,
93            nli: self.nli || other.nli,
94            classify: self.classify || other.classify,
95            stance: self.stance || other.stance,
96            token_counting: self.token_counting || other.token_counting,
97            local_inference: self.local_inference || other.local_inference,
98        }
99    }
100}

102impl From<&Capabilities> for HashSet<ModelCapability> {
103    fn from(caps: &Capabilities) -> Self {
104        let pairs = [
105            (caps.chat, ModelCapability::Chat),
106            (caps.chat_streaming, ModelCapability::ChatStreaming),
107            (caps.generate, ModelCapability::Generate),
108            (caps.tool_use, ModelCapability::ToolUse),
109            (caps.embed, ModelCapability::Embed),
110            (caps.nli, ModelCapability::Nli),
111            (caps.classify, ModelCapability::Classify),
112            (caps.stance, ModelCapability::Stance),
113            (caps.token_counting, ModelCapability::TokenCounting),
114            (caps.local_inference, ModelCapability::LocalInference),
115        ];
116        pairs
117            .into_iter()
118            .filter(|(enabled, _)| *enabled)
119            .map(|(_, cap)| cap)
120            .collect()
121    }
122}
