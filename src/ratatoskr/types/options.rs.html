7/// Options for chat requests (provider-agnostic)
8#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
9pub struct ChatOptions {
10    pub model: String,
11    #[serde(skip_serializing_if = "Option::is_none")]
12    pub temperature: Option<f32>,
13    #[serde(skip_serializing_if = "Option::is_none")]
14    pub max_tokens: Option<usize>,
15    #[serde(skip_serializing_if = "Option::is_none")]
16    pub top_p: Option<f32>,
17    #[serde(skip_serializing_if = "Option::is_none")]
18    pub top_k: Option<usize>,
19    #[serde(skip_serializing_if = "Option::is_none")]
20    pub stop: Option<Vec<String>>,
21    #[serde(skip_serializing_if = "Option::is_none")]
22    pub frequency_penalty: Option<f32>,
23    #[serde(skip_serializing_if = "Option::is_none")]
24    pub presence_penalty: Option<f32>,
25    #[serde(skip_serializing_if = "Option::is_none")]
26    pub seed: Option<u64>,
27    #[serde(skip_serializing_if = "Option::is_none")]
28    pub tool_choice: Option<ToolChoice>,
29    #[serde(skip_serializing_if = "Option::is_none")]
30    pub parallel_tool_calls: Option<bool>,
31    #[serde(skip_serializing_if = "Option::is_none")]
32    pub response_format: Option<ResponseFormat>,
33
34    // Normalized cross-provider options
35    #[serde(skip_serializing_if = "Option::is_none")]
36    pub cache_prompt: Option<bool>,
37    #[serde(skip_serializing_if = "Option::is_none")]
38    pub reasoning: Option<ReasoningConfig>,
39
40    // Escape hatch for truly provider-specific options
41    #[serde(skip_serializing_if = "Option::is_none")]
42    pub raw_provider_options: Option<serde_json::Value>,
43}

137    /// Returns the list of parameters that are set (not None) in these options.
138    ///
139    /// Used by the registry for validation against provider-declared parameters.
140    /// See also [`GenerateOptions::set_parameters`](crate::GenerateOptions::set_parameters).
141    pub fn set_parameters(&self) -> Vec<ParameterName> {
142        let mut params = Vec::new();
143        if self.temperature.is_some() {
144            params.push(ParameterName::Temperature);
145        }
146        if self.max_tokens.is_some() {
147            params.push(ParameterName::MaxTokens);
148        }
149        if self.top_p.is_some() {
150            params.push(ParameterName::TopP);
151        }
152        if self.top_k.is_some() {
153            params.push(ParameterName::TopK);
154        }
155        if self.stop.is_some() {
156            params.push(ParameterName::Stop);
157        }
158        if self.frequency_penalty.is_some() {
159            params.push(ParameterName::FrequencyPenalty);
160        }
161        if self.presence_penalty.is_some() {
162            params.push(ParameterName::PresencePenalty);
163        }
164        if self.seed.is_some() {
165            params.push(ParameterName::Seed);
166        }
167        if self.tool_choice.is_some() {
168            params.push(ParameterName::ToolChoice);
169        }
170        if self.parallel_tool_calls.is_some() {
171            params.push(ParameterName::ParallelToolCalls);
172        }
173        if self.response_format.is_some() {
174            params.push(ParameterName::ResponseFormat);
175        }
176        if self.cache_prompt.is_some() {
177            params.push(ParameterName::CachePrompt);
178        }
179        if self.reasoning.is_some() {
180            params.push(ParameterName::Reasoning);
181        }
182        params
183    }
184}

186/// Reasoning configuration for extended thinking models
187#[derive(Debug, Clone, Default, PartialEq, Eq, Serialize, Deserialize)]
188pub struct ReasoningConfig {
189    #[serde(skip_serializing_if = "Option::is_none")]
190    pub effort: Option<ReasoningEffort>,
191    #[serde(skip_serializing_if = "Option::is_none")]
192    pub max_tokens: Option<usize>,
193    #[serde(skip_serializing_if = "Option::is_none")]
194    pub exclude_from_output: Option<bool>,
195}
196
197/// Reasoning effort level
198///
199/// Matches OpenRouter's reasoning effort levels. Providers that support fewer
200/// levels (e.g. only low/medium/high) should map to the nearest equivalent.
201#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
202#[serde(rename_all = "lowercase")]
203#[non_exhaustive]
204pub enum ReasoningEffort {
205    None,
206    Minimal,
207    Low,
208    Medium,
209    High,
210    XHigh,
211}
