3use std::time::Duration;

5/// Ratatoskr error types
6#[derive(Debug, Clone, thiserror::Error)]
7#[non_exhaustive]
8pub enum RatatoskrError {
9    // Provider/network errors
10    #[error("HTTP error: {0}")]
11    Http(String),
12
13    #[error("API error ({status}): {message}")]
14    Api { status: u16, message: String },
15
16    /// Rate limited by the provider.
17    ///
18    /// `retry_after` is populated when the provider or gRPC status encodes a
19    /// duration. Currently `None` for errors originating from the `llm` crate
20    /// (which doesn't expose `Retry-After` headers yet); populated for
21    /// HuggingFace 429 responses and gRPC round-trips.
22    #[error("rate limited, retry after {retry_after:?}")]
23    RateLimited { retry_after: Option<Duration> },
24
25    #[error("authentication failed")]
26    AuthenticationFailed,
27
28    #[error("model not found: {0}")]
29    ModelNotFound(String),
30
31    #[error("preset not found: tier '{tier}', capability '{capability}'")]
32    PresetNotFound { tier: String, capability: String },
33
34    // Streaming errors
35    #[error("stream error: {0}")]
36    Stream(String),
37
38    // Data errors
39    #[error("JSON error: {0}")]
40    Json(String),
41
42    #[error("invalid input: {0}")]
43    InvalidInput(String),
44
45    // Configuration errors
46    #[error("no provider configured")]
47    NoProvider,
48
49    /// Provider cannot handle this model (wrong model, RAM constrained, etc.)
50    /// The registry should try the next provider in the fallback chain.
51    #[error("model not available from this provider")]
52    ModelNotAvailable,
53
54    #[error("configuration error: {0}")]
55    Configuration(String),
56
57    #[error("operation not implemented: {0}")]
58    NotImplemented(String),
59
60    #[error("provider does not support this operation")]
61    Unsupported,
62
63    /// Parameter is not supported by the target model/provider.
64    ///
65    /// Returned when validation policy is `Error` and a request contains
66    /// parameters the provider doesn't support.
67    #[error("unsupported parameter '{param}' for model '{model}' (provider: {provider})")]
68    UnsupportedParameter {
69        param: String,
70        model: String,
71        provider: String,
72    },
73
74    // Data processing errors
75    #[error("data error: {0}")]
76    DataError(String),
77
78    // Soft errors
79    #[error("empty response from model")]
80    EmptyResponse,
81
82    #[error("content filtered: {reason}")]
83    ContentFiltered { reason: String },
84
85    #[error("context length exceeded: {limit} tokens")]
86    ContextLengthExceeded { limit: usize },
87
88    // Wrapped llm crate error
89    #[error("LLM error: {0}")]
90    Llm(String),
91}
92
93impl RatatoskrError {
94    /// Whether this error is transient and the request may succeed on retry.
95    ///
96    /// Used by `RetryingProvider` to decide whether to retry a failed request.
97    /// Permanent errors (auth, validation, model not found) return `false`.
98    pub fn is_transient(&self) -> bool {
99        match self {
100            // Always transient
101            Self::RateLimited { .. } => true,
102            Self::Http(_) => true,
103            Self::Stream(_) => true,
104            Self::EmptyResponse => true,
105
106            // Server errors are transient; client errors are not
107            Self::Api { status, .. } => *status >= 500,
108
109            // Heuristic: network-sounding messages are transient
110            Self::Llm(msg) => {
111                let lower = msg.to_lowercase();
112                lower.contains("timeout")
113                    || lower.contains("connection")
114                    || lower.contains("reset")
115                    || lower.contains("refused")
116            }
117
118            // Everything else is permanent
119            Self::Json(_)
120            | Self::AuthenticationFailed
121            | Self::ModelNotFound(_)
122            | Self::PresetNotFound { .. }
123            | Self::InvalidInput(_)
124            | Self::NoProvider
125            | Self::ModelNotAvailable
126            | Self::Configuration(_)
127            | Self::NotImplemented(_)
128            | Self::Unsupported
129            | Self::UnsupportedParameter { .. }
130            | Self::DataError(_)
131            | Self::ContentFiltered { .. }
132            | Self::ContextLengthExceeded { .. } => false,
133        }
134    }
135
136    /// For `RateLimited` errors, the duration the provider suggests waiting.
137    ///
138    /// Returns `None` for non-rate-limit errors or when the provider didn't
139    /// specify a retry-after duration.
140    pub fn retry_after(&self) -> Option<Duration> {
141        match self {
142            Self::RateLimited { retry_after } => *retry_after,
143            _ => None,
144        }
145    }
146}

148impl From<serde_json::Error> for RatatoskrError {
149    fn from(err: serde_json::Error) -> Self {
150        RatatoskrError::Json(err.to_string())
151    }
152}
153impl From<llm::error::LLMError> for RatatoskrError {
154    fn from(err: llm::error::LLMError) -> Self {
155        // Map llm errors to our error types
156        let msg = err.to_string();
157        if msg.contains("rate limit") || msg.contains("429") {
158            RatatoskrError::RateLimited { retry_after: None }
159        } else if msg.contains("authentication")
160            || msg.contains("401")
161            || msg.contains("invalid api key")
162        {
163            RatatoskrError::AuthenticationFailed
164        } else if msg.contains("not found") || msg.contains("404") {
165            RatatoskrError::ModelNotFound(msg)
166        } else {
167            RatatoskrError::Llm(msg)
168        }
169    }
170}
