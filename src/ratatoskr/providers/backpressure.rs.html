1//! Streaming backpressure via bounded channels.
2//!
3//! Wraps stream output in a bounded `tokio::sync::mpsc::channel` so that
4//! producers block when consumers fall behind. Without this, a fast provider
5//! can fill unbounded memory if the consumer is slow (e.g. rate-limited
6//! client, busy UI thread).
7//!
8//! # Usage
9//!
10//! Applied automatically by [`ProviderRegistry`](super::ProviderRegistry)
11//! to `chat_stream` and `generate_stream` results. The buffer size defaults
12//! to [`DEFAULT_STREAM_BUFFER`] and can be overridden via the builder.
13
14use std::pin::Pin;
15
16use futures_util::{Stream, StreamExt};
17use tokio_stream::wrappers::ReceiverStream;
18
19use crate::Result;
20
21/// Default number of items buffered between producer and consumer.
22///
23/// 64 balances throughput (enough items to keep the consumer busy)
24/// with memory pressure (bounded, not unbounded).
25pub const DEFAULT_STREAM_BUFFER: usize = 64;
26
27/// Wrap a stream in a bounded channel for backpressure.
28///
29/// Spawns a producer task that reads from `inner` and sends items
30/// through a bounded `mpsc` channel. When the channel is full, the
31/// producer blocks until the consumer reads. If the consumer drops
32/// the stream, the producer stops.
33///
34/// # Panics
35///
36/// Requires a tokio runtime context (called within an async fn).
37pub fn bounded_stream<T: Send + 'static>(
38    inner: Pin<Box<dyn Stream<Item = Result<T>> + Send>>,
39    buffer_size: usize,
40) -> Pin<Box<dyn Stream<Item = Result<T>> + Send>> {
41    let (tx, rx) = tokio::sync::mpsc::channel(buffer_size);
42
43    tokio::spawn(async move {
44        let mut inner = inner;
45        while let Some(item) = inner.next().await {
46            if tx.send(item).await.is_err() {
47                break; // receiver dropped
48            }
49        }
50    });
51
52    Box::pin(ReceiverStream::new(rx))
53}
