1//! Ratatoskr - Unified model gateway for LLM APIs
2//!
3//! This crate provides a stable `ModelGateway` trait that abstracts over
4//! different LLM providers, allowing consumers to interact with models
5//! without coupling to provider-specific implementations.
6//!
7//! # Chat Example
8//!
9//! ```rust,no_run
10//! use ratatoskr::{Ratatoskr, Message, ChatOptions, ModelGateway};
11//!
12//! #[tokio::main]
13//! async fn main() -> ratatoskr::Result<()> {
14//!     let gateway = Ratatoskr::builder()
15//!         .openrouter(Some("sk-or-your-key"))
16//!         .build()?;
17//!
18//!     let response = gateway.chat(
19//!         &[
20//!             Message::system("You are a helpful assistant."),
21//!             Message::user("What is the capital of France?"),
22//!         ],
23//!         None,
24//!         &ChatOptions::new("anthropic/claude-sonnet-4"),
25//!     ).await?;
26//!
27//!     println!("{}", response.content);
28//!     Ok(())
29//! }
30//! ```
31//!
32//! # Embeddings Example (requires `huggingface` feature)
33//!
34//! ```rust,ignore
35//! use ratatoskr::{Ratatoskr, ModelGateway};
36//!
37//! #[tokio::main]
38//! async fn main() -> ratatoskr::Result<()> {
39//!     let gateway = Ratatoskr::builder()
40//!         .huggingface("hf_your_key")
41//!         .build()?;
42//!
43//!     let embedding = gateway.embed(
44//!         "Hello, world!",
45//!         "sentence-transformers/all-MiniLM-L6-v2",
46//!     ).await?;
47//!
48//!     println!("Dimensions: {}", embedding.dimensions);
49//!     Ok(())
50//! }
51//! ```
52//!
53//! # Text Generation Example
54//!
55//! ```rust,ignore
56//! use ratatoskr::{Ratatoskr, GenerateOptions, ModelGateway};
57//!
58//! #[tokio::main]
59//! async fn main() -> ratatoskr::Result<()> {
60//!     let gateway = Ratatoskr::builder()
61//!         .ollama("http://localhost:11434")
62//!         .build()?;
63//!
64//!     let response = gateway.generate(
65//!         "Once upon a time",
66//!         &GenerateOptions::new("llama3.2:1b").max_tokens(100),
67//!     ).await?;
68//!
69//!     println!("{}", response.text);
70//!     Ok(())
71//! }
72//! ```
