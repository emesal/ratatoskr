13/// The core gateway trait that all implementations must provide.
14///
15/// This trait abstracts over different LLM providers, allowing consumers
16/// to interact with models without coupling to provider-specific implementations.
17#[async_trait]
18pub trait ModelGateway: Send + Sync {
19    // ===== Phase 1: Chat (must implement) =====
20
21    /// Streaming chat completion
22    async fn chat_stream(
23        &self,
24        messages: &[Message],
25        tools: Option<&[ToolDefinition]>,
26        options: &ChatOptions,
27    ) -> Result<Pin<Box<dyn Stream<Item = Result<ChatEvent>> + Send>>>;
28
29    /// Non-streaming chat completion
30    async fn chat(
31        &self,
32        messages: &[Message],
33        tools: Option<&[ToolDefinition]>,
34        options: &ChatOptions,
35    ) -> Result<ChatResponse>;
36
37    /// What can this gateway do?
38    fn capabilities(&self) -> Capabilities;
39
40    // ===== Phase 3+: Default stubs =====
41
42    /// Generate embeddings for text
43    async fn embed(&self, _text: &str, _model: &str) -> Result<Embedding> {
44        Err(RatatoskrError::NotImplemented("embed".into()))
45    }
46
47    /// Batch embedding generation
48    async fn embed_batch(&self, _texts: &[&str], _model: &str) -> Result<Vec<Embedding>> {
49        Err(RatatoskrError::NotImplemented("embed_batch".into()))
50    }
51
52    /// Natural language inference
53    async fn infer_nli(
54        &self,
55        _premise: &str,
56        _hypothesis: &str,
57        _model: &str,
58    ) -> Result<NliResult> {
59        Err(RatatoskrError::NotImplemented("infer_nli".into()))
60    }
61
62    /// Zero-shot classification
63    async fn classify_zero_shot(
64        &self,
65        _text: &str,
66        _labels: &[&str],
67        _model: &str,
68    ) -> Result<ClassifyResult> {
69        Err(RatatoskrError::NotImplemented("classify_zero_shot".into()))
70    }
71
72    /// Count tokens for a given model
73    fn count_tokens(&self, _text: &str, _model: &str) -> Result<usize> {
74        Err(RatatoskrError::NotImplemented("count_tokens".into()))
75    }
76
77    /// Batch NLI inference — more efficient for multiple pairs
78    async fn infer_nli_batch(&self, pairs: &[(&str, &str)], model: &str) -> Result<Vec<NliResult>> {
79        // Default: sequential fallback
80        let mut results = Vec::with_capacity(pairs.len());
81        for (premise, hypothesis) in pairs {
82            results.push(self.infer_nli(premise, hypothesis, model).await?);
83        }
84        Ok(results)
85    }
86
87    /// Non-streaming text generation
88    async fn generate(
89        &self,
90        _prompt: &str,
91        _options: &GenerateOptions,
92    ) -> Result<GenerateResponse> {
93        Err(RatatoskrError::NotImplemented("generate".into()))
94    }
95
96    /// Streaming text generation
97    async fn generate_stream(
98        &self,
99        _prompt: &str,
100        _options: &GenerateOptions,
101    ) -> Result<Pin<Box<dyn Stream<Item = Result<GenerateEvent>> + Send>>> {
102        Err(RatatoskrError::NotImplemented("generate_stream".into()))
103    }
104
105    // ===== Phase 5: Extended capabilities =====
106
107    /// Stance detection toward a target topic.

109    /// Determines whether text expresses favor, against, or neutral stance

111    async fn classify_stance(
112        &self,
113        _text: &str,
114        _target: &str,
115        _model: &str,

117        Err(RatatoskrError::NotImplemented("classify_stance".into()))
118    }
119
120    /// Tokenize text into detailed Token objects with IDs, text, and byte offsets.
121    fn tokenize(&self, _text: &str, _model: &str) -> Result<Vec<Token>> {
122        Err(RatatoskrError::NotImplemented("tokenize".into()))
123    }
124
125    /// List all available models and their capabilities.
126    fn list_models(&self) -> Vec<ModelInfo> {
127        vec![]
128    }
129
130    /// Get the status of a specific model.
131    fn model_status(&self, _model: &str) -> ModelStatus {
132        ModelStatus::Unavailable {
133            reason: "Not implemented".into(),

135    }
136
137    // ===== Phase 6: Model intelligence =====
138
139    /// Get extended metadata for a model, including parameter availability.
140    ///
141    /// Returns `None` if the model is not known to the registry or cache.
142    /// This is a synchronous lookup — see [`fetch_model_metadata`](Self::fetch_model_metadata)
143    /// for the async variant that can reach out to provider APIs.
144    fn model_metadata(&self, _model: &str) -> Option<ModelMetadata> {
145        None
146    }
147
148    /// Fetch metadata from the provider that would serve this model.
149    ///

151    /// and returns the result. Use [`model_metadata`](Self::model_metadata) for
152    /// cached/registry lookups without network I/O.
153    async fn fetch_model_metadata(&self, _model: &str) -> Result<ModelMetadata> {
154        Err(RatatoskrError::NotImplemented(
155            "fetch_model_metadata".into(),
156        ))
157    }

159    // ===== Autoconfig presets =====
160
161    /// Resolve a preset for the given cost tier and capability.
162    ///
163    /// Returns the concrete model ID and any default generation parameters,
164    /// or `None` if no preset is configured for this combination.
165    fn resolve_preset(&self, _tier: &str, _capability: &str) -> Option<crate::PresetResolution> {
166        None
167    }
168
169    /// List all configured presets as a tier → capability set map.
170    ///
171    /// Returns the full two-level structure of preset keys so consumers can
172    /// discover which tiers and capabilities are available before resolving.
173    /// Use [`Self::resolve_preset`] to obtain the concrete model and parameters.
174    fn list_presets(
175        &self,
176    ) -> std::collections::BTreeMap<String, std::collections::BTreeSet<String>> {
177        Default::default()
178    }
