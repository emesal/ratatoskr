172    /// Set the parameter validation policy.
173    ///
174    /// Controls how the registry handles requests containing parameters
175    /// not declared as supported by the provider:
176    ///
177    /// - [`ParameterValidationPolicy::Warn`] — log a warning, proceed with request (default)
178    /// - [`ParameterValidationPolicy::Error`] — return `UnsupportedParameter` error
179    /// - [`ParameterValidationPolicy::Ignore`] — silently proceed
180    pub fn validation_policy(mut self, policy: ParameterValidationPolicy) -> Self {
181        self.validation_policy = policy;
182        self
183    }

185    /// Check if at least one chat provider is configured.
186    fn has_chat_provider(&self) -> bool {
187        self.openrouter_key.is_some()
188            || self.anthropic_key.is_some()
189            || self.openai_key.is_some()
190            || self.google_key.is_some()
191            || self.ollama_url.is_some()
192    }
193
194    /// Check if at least one capability provider is configured.
195    fn has_capability_provider(&self) -> bool {
196        #[cfg(feature = "huggingface")]
197        if self.huggingface_key.is_some() {
198            return true;
199        }
200        #[cfg(feature = "local-inference")]
201        if self.local_embedding_model.is_some() || self.local_nli_model.is_some() {
202            return true;
203        }
204        false
205    }

207    /// Build the gateway.
208    pub fn build(self) -> Result<EmbeddedGateway> {
209        use crate::cache::ModelCache;
210        use crate::providers::{LlmChatProvider, ProviderRegistry};
211        use llm::builder::LLMBackend;
212        use std::sync::Arc;
213
214        // Must have at least one provider (chat or capability)
215        if !self.has_chat_provider() && !self.has_capability_provider() {
216            return Err(RatatoskrError::NoProvider);
217        }
218
219        let timeout_secs = self.default_timeout_secs.unwrap_or(120);
220
221        // Build model manager for local inference
222        #[cfg(feature = "local-inference")]
223        let model_manager = {
224            use crate::model::ModelManagerConfig;
225            let config = ModelManagerConfig {
226                cache_dir: self.cache_dir.unwrap_or_else(|| {
227                    std::env::var("RATATOSKR_CACHE_DIR")
228                        .map(std::path::PathBuf::from)
229                        .unwrap_or_else(|_| {
230                            dirs::cache_dir()
231                                .unwrap_or_else(|| std::path::PathBuf::from(".cache"))
232                                .join("ratatoskr")
233                                .join("models")
234                        })
235                }),
236                default_device: self.device.clone(),
237                ram_budget: self.ram_budget,
238            };
239            Arc::new(crate::model::ModelManager::new(config))
240        };
241
242        // Build provider registry with fallback chain
243        let mut registry = ProviderRegistry::new();
244        registry.set_validation_policy(self.validation_policy);
245
246        // =====================================================================
247        // Register LOCAL providers FIRST (higher priority)
248        // =====================================================================
249
250        #[cfg(feature = "local-inference")]
251        if let Some(model) = &self.local_embedding_model {
252            let provider = Arc::new(crate::providers::LocalEmbeddingProvider::new(
253                model.clone(),
254                model_manager.clone(),
255            ));
256            registry.add_embedding(provider);
257        }
258
259        #[cfg(feature = "local-inference")]
260        if let Some(model) = &self.local_nli_model {
261            let provider = Arc::new(crate::providers::LocalNliProvider::new(
262                model.clone(),
263                model_manager.clone(),
264            ));
265            registry.add_nli(provider);
266        }
267
268        // =====================================================================
269        // Register API providers as FALLBACKS (lower priority)
270        // =====================================================================
271
272        #[cfg(feature = "huggingface")]
273        if let Some(ref key) = self.huggingface_key {
274            let client = Arc::new(crate::providers::HuggingFaceClient::new(key));
275
276            // HuggingFace accepts any model, registered as fallback
277            registry.add_embedding(client.clone());
278            registry.add_nli(client.clone());
279            registry.add_classify(client.clone());
280
281            // Also register ZeroShotStanceProvider as stance fallback
282            let stance_fallback = Arc::new(crate::providers::ZeroShotStanceProvider::new(
283                client.clone(),
284                "facebook/bart-large-mnli",
285            ));
286            registry.add_stance(stance_fallback);
287        }
288
289        // =====================================================================
290        // Register CHAT providers (shared HTTP client for metadata fetches)
291        // =====================================================================
292
293        let http_client = reqwest::Client::new();
294
295        // Helper: build an LlmChatProvider with the shared http client
296        let make_provider = |backend, key: String, name: &str| -> Arc<LlmChatProvider> {
297            Arc::new(
298                LlmChatProvider::with_http_client(backend, key, name, http_client.clone())
299                    .timeout_secs(timeout_secs),
300            )
301        };
302
303        // OpenRouter (routes to many models, good default)
304        if let Some(ref key) = self.openrouter_key {
305            let provider = make_provider(LLMBackend::OpenRouter, key.clone(), "openrouter");
306            registry.add_chat(provider.clone());
307            registry.add_generate(provider);
308        }
309
310        // Direct Anthropic
311        if let Some(ref key) = self.anthropic_key {
312            let provider = make_provider(LLMBackend::Anthropic, key.clone(), "anthropic");
313            registry.add_chat(provider.clone());
314            registry.add_generate(provider);
315        }
316
317        // Direct OpenAI
318        if let Some(ref key) = self.openai_key {
319            let provider = make_provider(LLMBackend::OpenAI, key.clone(), "openai");
320            registry.add_chat(provider.clone());
321            registry.add_generate(provider);
322        }
323
324        // Google (Gemini)
325        if let Some(ref key) = self.google_key {
326            let provider = make_provider(LLMBackend::Google, key.clone(), "google");
327            registry.add_chat(provider.clone());
328            registry.add_generate(provider);
329        }
330
331        // Ollama
332        if let Some(ref url) = self.ollama_url {
333            let provider = Arc::new(
334                LlmChatProvider::with_http_client(
335                    LLMBackend::Ollama,
336                    "ollama",
337                    "ollama",
338                    http_client.clone(),
339                )
340                .timeout_secs(timeout_secs)
341                .ollama_url(url.clone()),
342            );
343            registry.add_chat(provider.clone());
344            registry.add_generate(provider);
345        }
346
347        // Build tokenizer registry for local inference
348        #[cfg(feature = "local-inference")]
349        let tokenizer_registry = {
350            let mut registry = crate::tokenizer::TokenizerRegistry::new();
351            for (pattern, source) in self.tokenizer_mappings {
352                registry.register(&pattern, source);
353            }
354            Arc::new(registry)
355        };
356
357        let model_registry = crate::registry::ModelRegistry::with_embedded_seed();
358        let model_cache = Arc::new(ModelCache::new());
359
360        Ok(EmbeddedGateway::new(
361            registry,
362            model_registry,
363            model_cache,
364            #[cfg(feature = "local-inference")]
365            model_manager,
366            #[cfg(feature = "local-inference")]
367            tokenizer_registry,
368        ))
369    }
370}
